import csv

def isCoprime(a, b):
    '''计算两个数是否互质'''
    if a > b:
        a, b = b, a
    while a != 0:
        a, b = b % a, a
    return b == 1

def getMaxValue(maxRetry):
    '''获取用户希望搜索勾股数的范围，默认从1开始'''

    flag = 1
    info = "请输入搜索最大值（例如 10000）："
    maxValue = -1
    while flag and maxRetry > 0:
        try:
            maxValue = int(input(info))
        except Exception as e:
            info = "输入数值格式错误，请重新输入："
        else:
            flag = 0
            break
        maxRetry -= 1
    #如果尝试三次均未成功或者输入的值非正则推出
    if flag == 1 or maxValue <= 0:
        exit(1)
    return maxValue
